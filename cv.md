function largest (n){
  if (n <= 0 || !Number.isInteger(n)) {
    return false;
  }
  let p = 10 ** n - 1;
  let i = 2;
  while (i < p){
    if (p % i){ 
      i += 1;
    }
    else{
      i = 2, p -= 1;
    }
 }
  return p;
}
